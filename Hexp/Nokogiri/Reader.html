def call(node)
  return node.text if node.text?

  unless node.attributes.empty?
    attrs = node.attributes.map do |key, value|
      [key.to_sym, value.value]
    end
    attrs = Hash[attrs]
  end

  recurse = ->(next_node) { call(next_node) }
  H[node.name.to_sym, attrs, node.children.map(&recurse)]
end
